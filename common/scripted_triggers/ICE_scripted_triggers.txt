ICE_allied_weapons_seller_trigger = {
	custom_trigger_tooltip = {
		tooltip = "TRIGGER_ANY_ALLIED_COUNTRY_STARTS" #Alright it's technically not true
		FROM = {
			NOT = { tag = ROOT}
			OR = {
				is_in_faction_with = ROOT
				AND = {
					is_in_faction = no
					has_war = no
				}
			}
		}
	}
}
