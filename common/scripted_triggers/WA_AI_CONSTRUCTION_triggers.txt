WA_AI_PC_can_afford_project = {

	@AI_MAX_FRACTION_OF_FACTORIES_TO_ASSIGN_ON_PROJECTS_TOTAL = 0.35
	@AI_MIN_NUM_OF_FACTORIES_TO_ASSIGN_ON_PROJECT = 20

	# check that the ai can afford to assign factories to a new project
	set_temp_variable = { max_factories_on_projects = ROOT.num_of_civilian_factories_available_for_projects }
	add_to_temp_variable = { max_factories_on_projects = ROOT.WA_AI_PC_assigned_factories_total }
	multiply_temp_variable = { max_factories_on_projects = @AI_MAX_FRACTION_OF_FACTORIES_TO_ASSIGN_ON_PROJECTS_TOTAL }

	set_temp_variable = { min_factories_on_projects = ROOT.WA_AI_PC_assigned_factories_total }
	add_to_temp_variable = { min_factories_on_projects = @AI_MIN_NUM_OF_FACTORIES_TO_ASSIGN_ON_PROJECT }

	check_variable = { min_factories_on_projects < max_factories_on_projects }
}

WA_AI_PC_can_build_project = {
	OR = {
		is_controlled_by = ROOT
		CONTROLLER = {
			is_subject_of = ROOT
		}
	}


	free_building_slots = { building = infrastructure size < 10 include_locked = no } # not impassable

	# is not already building the same type of building here
	if = { limit = { is_in_array = { THIS.WA_AI_PC_projects_in_state = ROOT.id } }

		NOT = { check_variable = { ROOT.WA_AI_PC_project_building_type^THIS.WA_AI_PC_project_target_state_@ROOT = ROOT.WA_AI_PC_project_building_type^-1 } }
	}

	if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 1 } }

		free_building_slots = { building = infrastructure size > 0 include_locked = no }
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 2 } }

		free_building_slots = { building = air_base size > 0 include_locked = no }
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 3 } }

		free_building_slots = { building = anti_air_building size > 0 include_locked = no }
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 4 } }

		OR = {
			AND = { ROOT = { has_tech = radio_detection } 				radar_station < 1 }
			AND = { ROOT = { has_tech = decimetric_radar } 				radar_station < 2 }
			AND = { ROOT = { has_tech = improved_decimetric_radar } 	radar_station < 3 }
			AND = { ROOT = { has_tech = centimetric_radar } 			radar_station < 4 }
			AND = { ROOT = { has_tech = improved_centimetric_radar } 	radar_station < 5 }
			AND = { ROOT = { has_tech = advanced_centimetric_radar } 	radar_station < 6 }
		}
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 5 } }

		free_building_slots = { building = arms_factory size > 0 include_locked = no }
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 6 } }

		free_building_slots = { building = industrial_complex size > 0 include_locked = no }
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 7 } }

		free_building_slots = { building = dockyard size > 0 include_locked = no }
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 8 } }

		OR = {
			free_building_slots = { building = synthetic_refinery size > 0 include_locked = no }
			free_building_slots = { building = industrial_complex size > 0 include_locked = no }
			free_building_slots = { building = arms_factory size > 0 include_locked = no }
			free_building_slots = { building = dockyard size > 0 include_locked = no }
		}
		synthetic_refinery < 3
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 9 } }

		check_variable = { building_level@arms_factory > 0 }
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 10 } }

		check_variable = { building_level@industrial_complex > 0 }
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 11 } }

		free_building_slots = { building = hydro_steel_refinery size > 0 include_locked = no }
	}

	else_if = { limit = { check_variable = { ROOT.WA_AI_PC_project_building_type^-1 = 12 } }

		free_building_slots = { building = hydro_aluminium_refinery size > 0 include_locked = no }
	}
}

### air base

WA_AI_airbase_priority = {
	OR = {
		AND = { # USA, JAP
			ROOT = { OR = { original_tag = USA original_tag = JAP } }
			air_base < 4
			is_island_state = yes
			OR = { is_on_continent = asia is_on_continent = pacific }
		}

		AND = { # GER
			ROOT = { original_tag = GER }
			SOV = { WA_AI_is_threat = yes}
			air_base < 4
			any_neighbor_state = { CONTROLLER = { original_tag = SOV } }
			OR = { is_on_continent = europe }
		}

		AND = { # SOV
			ROOT = { original_tag = SOV }
			GER = { WA_AI_is_threat = yes}
			air_base < 4
			any_neighbor_state = { CONTROLLER = { original_tag = GER } }
			OR = { is_on_continent = europe }
		}
	}
}

### Supply hub detection

# THIS = STATE - Check if state has a supply hub (supply_node or naval_base)
WA_AI_PC_state_has_supply_hub = {
	OR = {
		any_province_building_level = {
			province = { all_provinces = yes }
			building = supply_node
			level > 0
		}
		any_province_building_level = {
			province = { all_provinces = yes }
			building = naval_base
			level > 0
		}
	}
}

# _province_hub_id - Check if specific province has a supply hub
WA_AI_PC_prov_has_supply_hub = {
	meta_trigger = {
		text = {
			OR = {
				any_province_building_level = {
					province = { id = [x] }
					building = supply_node
					level > 0
				}
				any_province_building_level = {
					province = { id = [x] }
					building = naval_base
					level > 0
				}
			}
		}
		x = "[?_province_hub_id]"
	}
}

# THIS = STATE, _check_railway_level - Check if state already has railway at specified level
# Uses meta_trigger to allow variable level parameter

	meta_trigger = {
		text = {
			has_railway_level = {
				state = THIS
				level = [x]
			}
		}
		x = "[?_check_railway_level]"
	}
}
