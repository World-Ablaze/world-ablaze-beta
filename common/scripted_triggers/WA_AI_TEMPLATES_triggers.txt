############################################################################################################
# WA AI Templates Triggers
# Templates types :
# - Infantry
# - Motorized
# - Mechanized
# - Light Armor
# - Medium Armor
# - Heavy Armor
# - Modern Armor
# - Garrison
# - Military Police / Cavalry
# - Marine
# - Airborne
# - Mountaineer
# - Rangers


# Available widths :
# - 20 Width
# - 30 Width

# parameters :
# - fuel availability
# - total manpower
# - mils count
# - expected terrain

WA_AI_TEMPLATES_use_heavy_infantry = {
	num_of_military_factories > 5
}

WA_AI_TEMPLATES_use_marines = {
	any_owned_state = {
		is_coastal = yes
	}
	num_of_military_factories > 50
	NOT = { WA_AI_CONFIG_is_submarine_navy = yes }
	NOT = { WA_AI_CONFIG_DIVISIONS_is_expected_marshes_or_mountains = yes }
}

WA_AI_TEMPLATES_use_mountaineers = {
	NOT = { WA_AI_TEMPLATES_use_marines = yes }
}

WA_AI_TEMPLATES_use_motorized = {
	WA_AI_CONFIG_DIVISIONS_use_mechanized_divisions = yes
	num_of_military_factories < 101
}

WA_AI_TEMPLATES_use_mechanized = {
	OR = {
		WA_AI_CONFIG_DIVISIONS_use_mechanized_divisions = yes
		num_of_military_factories > 100
	}
}

WA_AI_TEMPLATES_use_armor = {
	OR = {
		WA_AI_CONFIG_DIVISIONS_use_armored_divisions = yes
		num_of_military_factories > 150
	}
}

WA_AI_TEMPLATES_use_heavy_armor = {
	WA_AI_TEMPLATES_use_armor = yes
	WA_AI_CONFIG_DIVISIONS_focus_on_heavy_armor = yes
}

WA_AI_TEMPLATES_use_medium_armor = {
	WA_AI_TEMPLATES_use_armor = yes
	NOT = { WA_AI_TEMPLATES_use_heavy_armor = yes }
	OR = {
		WA_AI_CONFIG_DIVISIONS_focus_on_medium_armor = yes
		date > 1941.1.1
	}
}

WA_AI_TEMPLATES_use_light_armor = {
	WA_AI_TEMPLATES_use_armor = yes
}

WA_AI_TEMPLATES_use_modern_armor = {
	WA_AI_TEMPLATES_use_armor = yes
	date > 1945.1.1
}
