		is_literally_china = yes

GER_will_support = {
	GER = {
		exists = yes
		has_capitulated = no
		NOT = {
			has_war_with = ROOT
		}
		OR = {
			has_opinion = { target = ROOT value > -1 }
			is_in_faction_with = ROOT
			ROOT = {
				is_subject_of = GER
			}
		}
	}
}
