		original_tag = BUL
		has_dlc = "Battle for the Bosporus"
	}
	enable = {
		focus_progress = {
			focus = BUL_bulgarian_irredentism
			progress > 0.1
		}
		country_exists = GER
		GER = {
			has_opinion	= {
				target = ROOT
				value < 50
			}
			has_capitulated = no
			is_subject = no
		}
		OR = {
			has_government = neutrality
			has_government = fascism
		}

	abort = {
		OR = {
			NOT = { country_exists = GER }

						target = ROOT
						value > 80
					}
					has_capitulated = yes
					is_subject = yes
				}
			}
		}
	}

	ai_strategy = {
		type = diplo_action_desire
		id = GER
		target = improve_relation
		value = 200
	}
}
