

build_intelligence_agency_as_major = {
	enable = {
		is_major = yes
		OR = {
			has_war = yes
			date > 1936.1.1
		}
		NOT = { has_idea = undisturbed_isolation }
		NOT = { has_idea = isolation }
		NOT = { has_idea = civilian_economy }
	}

	ai_strategy = {
		type = intelligence_agency_usable_factories
		value = 10
	}

}

build_intelligence_agency = {
	enable = {
		date > 1938.6.1
		has_war = no
		num_of_civilian_factories > 50
		NOT = { has_idea = undisturbed_isolation }
		NOT = { has_idea = isolation }
		NOT = { has_idea = civilian_economy }
	}
	abort = {
		OR = {
			has_war = yes
			num_of_civilian_factories < 50
		}
	}

	ai_strategy = {
		type = intelligence_agency_usable_factories
		value = 10
	}

}

upgrade_intelligence_agency = {
	enable = {
		date > 1939.6.1
		num_of_civilian_factories > 100
	}
	abort = {
		OR = {
			num_of_civilian_factories < 80
		}
	}

	ai_strategy = {
		type = intelligence_agency_usable_factories
		value = 20
	}
}

upgrade_intelligence_agency_as_spy_master = {
	enable = {

		num_of_civilian_factories < 140
	}

	ai_strategy = {
		type = intelligence_agency_usable_factories
		value = 60
	}
}

ask_foreign_manpower_as_major = {
	enable = {
		is_major = yes
		garrison_manpower_need > 200000
	}
	
	abort = {
		garrison_manpower_need < 100000
	}

	ai_strategy = {
		type = asking_foreign_garrison
		value = 100
		# id = ITA # id is optionnal. If it is filled, the value will just apply for this country. If not, the value will apply for all countries.
	}
}
