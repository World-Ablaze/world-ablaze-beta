DEFAULT_market_stockpile_management = {
    enable = {
		OR = {
			AND = {
				has_war = yes
				OR = {
					tag = FRA
					tag = CZE
					tag = SWE
				}
			}
			NOT = {
				tag = FRA
				tag = CZE
				tag = SWE
			}
		}
        OR = {
            AND = {
                tag = USA
                USA = {
                    OR = {
                        NOT = { has_completed_focus = USA_cash_and_carry_act }
                        has_completed_focus = USA_lend_lease_act
                    }
                }
            }
            NOT = {
                tag = USA
            }
        }
