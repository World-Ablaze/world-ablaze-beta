# ============================================================================
# Tank Equipment Production Strategies
# ============================================================================
# These strategies are used to control the production of tank equipment.
# All strategies use triggers from WA_AI_PRODUCTION_tanks.txt
# No country tags should appear in this file.
#
# Main strategy types used:
# - equipment_production_min_factories_archetype: minimum factories for archetype
# - equipment_variant_production_factor: production factor for specific variant
# ============================================================================

# ============================================================================
# LIGHT ARMOR
# ============================================================================

WA_AI_PRODUCTION_build_light_armor = {
	enable = { WA_AI_PRODUCTION_build_light_armor = yes }
	abort_when_not_enabled = yes

	ai_strategy = { type = equipment_variant_production_factor id = light_tank_chassis value = 30 }
}
