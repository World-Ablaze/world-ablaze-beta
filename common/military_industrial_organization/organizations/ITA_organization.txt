ITA_fiat_organization = {
	include = generic_tank_organization
	icon = GFX_idea_fiat
	allowed = {	
		OR = {
			tag = ITA 
			AND = {
				tag = BUL 
				has_dlc = "Battle for the Bosporus"
			}
			AND = {
				tag = ETH 
				has_dlc = "By Blood Alone"
			}
		}
	}
	available = {
		IF = {
			limit = { owner = { original_tag = ITA } }
			OWNER = {
				controls_state = 158 #PIEMONT (Torino)
				
			}
			ITA_is_northern_designer = yes
		}
		IF = {
			limit = { owner = { original_tag = BUL } }
			custom_trigger_tooltip = {
				tooltip = has_invited_mio_tt
				owner = {
					has_country_flag = has_invited_fiat_mio_flag
				}
			}
			owner = {
				OR = {
					is_subject = no
					is_subject_of = ITA
				}
				is_in_faction_with = ITA
				NOT = { has_completed_focus = BUL_nationalization }
			}
		}
	}
	visible = {
		IF = {
			limit = { owner = { original_tag = ETH } }
			owner = { has_country_flag = ETH_ITA_sends_army_designers }
		}
	}
}
