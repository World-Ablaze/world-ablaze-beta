GER_henschel_organization = {
	include = generic_heavy_tank_organization
	icon = GFX_idea_henschel

	allowed = {
		OR = {
			tag = GER
			AND = {
				tag = BUL
				has_dlc = "Battle for the Bosporus"
			}
		}
	}
	available = {
		FROM = {NOT = { has_country_flag = henschel_lost }}
		IF = {
			limit = { owner = { original_tag = BUL } }
			custom_trigger_tooltip = {
				tooltip = has_invited_mio_tt
				owner = {
					has_country_flag = has_invited_henschel_mio_flag
				}
			}
		}
		IF = {
			limit = { owner = { original_tag = BUL } }
			owner = {
				OR = {
					is_subject = no
					is_subject_of = GER
				}
				is_in_faction_with = GER
				NOT = { has_completed_focus = BUL_nationalization }
			}
		}
	}
}
