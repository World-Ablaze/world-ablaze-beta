scripted_gui = {

    WA_reopen_welcome_ui = {
        context_type = player_context

		window_name = "WA_reopen_welcome_screen_container"

        effects = {
			wa_reopen_welcome_screen_click = {
				set_variable = { wa_welcome_screen_state = 0 }
			}
        }
    }

	WA_welcome_ui = {
		context_type = player_context

		window_name = "WA_welcome_window"

		visible = {
			check_variable = { wa_welcome_screen_state > -1 }
		}

		properties = {
            economy_changes_tab = {
				frame = economy_tab_state
			}
			tech_changes_tab = {
				frame = tech_tab_state
			}
			combat_changes_tab = {
				frame = combat_tab_state
			}
		}

		triggers = {
            economy_changes_tab_click_enabled = {
                if = {
                    limit = {
                        check_variable = { wa_welcome_screen_state = 1 }
                    }
                    set_temp_variable = { economy_tab_state = 1 }
                } else_if = {
                    limit = {
                        check_variable = { wa_welcome_screen_state > -1 }
                    }
                    set_temp_variable = { economy_tab_state = 2 }
                }
            }
            tech_changes_tab_click_enabled = {
                if = {
                    limit = {
                        check_variable = { wa_welcome_screen_state = 2 }
                    }
                    set_temp_variable = { tech_tab_state = 1 }
                } else_if = {
                    limit = {
                        check_variable = { wa_welcome_screen_state > -1 }
                    }
                    set_temp_variable = { tech_tab_state = 2 }
                }
            }
            combat_changes_tab_click_enabled = {
                if = {
                    limit = {
                        check_variable = { wa_welcome_screen_state = 3 }
                    }
                    set_temp_variable = { combat_tab_state = 1 }
                } else_if = {
                    limit = {
                        check_variable = { wa_welcome_screen_state > -1 }
                    }
                    set_temp_variable = { combat_tab_state = 2 }
                }
            }
		}

		effects = {
			exit_button_click = {
				set_variable = { wa_welcome_screen_state = -1 }
			}
            economy_changes_tab_click = {
                if = {
                    limit = {
                        check_variable = { wa_welcome_screen_state = 1 }
                    }
                    set_variable = { wa_welcome_screen_state = 0 }
                } else = {
                    set_variable = { wa_welcome_screen_state = 1 }
                }
            }
            tech_changes_tab_click = {
                if = {
                    limit = {
                        check_variable = { wa_welcome_screen_state = 2 }
                    }
                    set_variable = { wa_welcome_screen_state = 0 }
                } else = {
                    set_variable = { wa_welcome_screen_state = 2 }
                }
            }
            combat_changes_tab_click = {
                if = {
                    limit = {
                        check_variable = { wa_welcome_screen_state = 3 }
                    }
                    set_variable = { wa_welcome_screen_state = 0 }
                } else = {
                    set_variable = { wa_welcome_screen_state = 3 }
                }
            }
		}
	}
}
