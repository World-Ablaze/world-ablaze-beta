# Scripted effects to prevent AI from producing certain ships depending on difficulty setting.
# Historical mode: blocks all capital ship archetypes for all major naval powers. as KDE system is used
# Competitive mode: blocks individually identified 'bad' ships per country.
# Triggered at game start via on_actions or country_scripted_effects.

###############################################
# UNITED STATES
###############################################

WA_AI_NAVY_exclusions_USA = {
	# --- HISTORICAL MODE: block all capital archetypes ---
	if = {
		limit = { WA_AI_DIFFICULTY_is_historical = yes }

		add_ai_strategy = {
			type = equipment_variant_production_factor
			id = "battleship"
			value = -200
		}
		add_ai_strategy = {
			type = equipment_variant_production_factor
			id = "ship_hull_heavy"
			value = -200
		}
		add_ai_strategy = {
			type = equipment_variant_production_factor
			id = "carrier"
			value = -200
		}
		add_ai_strategy = {
			type = equipment_variant_production_factor
			id = "ship_hull_heavy_crusier"
			value = -200
		}
	}

	# --- COMPETITIVE MODE: block specific bad ships ---
	if = {
		limit = { WA_AI_DIFFICULTY_is_competitive = yes }

		add_ai_strategy = {
			type = equipment_variant_production_factor
			id = "usa_carrier_hull_5"
			value = -200
		}
	}
}
