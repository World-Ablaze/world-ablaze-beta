GER_garrison_control_tooltip = {
	if = {
		limit = {
			check_variable = { GER_garrison_control = 1 }
			ROOT = { has_completed_focus = GER_prepare_rebel_strongholds }
		}
		custom_effect_tooltip = GER_garrison_control_none_tt
	}
	if = {
		limit = {
			check_variable = { GER_garrison_control = 2 }
			ROOT = { has_completed_focus = GER_prepare_rebel_strongholds }
		}
		custom_effect_tooltip = GER_garrison_control_weak_tt
	}
	if = {
		limit = {
			check_variable = { GER_garrison_control = 3 }
			ROOT = { has_completed_focus = GER_prepare_rebel_strongholds }
		}
		custom_effect_tooltip = GER_garrison_control_strong_tt
	}
	if = {
		limit = {
			check_variable = { GER_garrison_control = 4 }
			ROOT = { has_completed_focus = GER_prepare_rebel_strongholds }
		}
		custom_effect_tooltip = GER_garrison_control_total_tt
	}
}

GER_garrison_control_cost_calculation = {
	if = {
		limit = {
			check_variable = { GER_garrison_control > 1 }
		}
		set_variable = { GER_garrison_control_cost = GER_garrison_control }
		multiply_variable = { GER_garrison_control_cost = 10 }
	}
}
