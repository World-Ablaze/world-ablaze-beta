
mst_reformat_variables = {
	set_variable = { global.mst_display_variable_C1 = global.mst_variable_C1 }
	set_variable = { global.mst_display_variable_C2 = global.mst_variable_C2 }
	divide_variable = { global.mst_display_variable_C1 = 1000 }
	divide_variable = { global.mst_display_variable_C2 = 1000 }
}

bst_count_buildings = {
	bst_count_civf = yes
	bst_count_milf = yes
	bst_count_navf = yes
	bst_count_silo = yes
	bst_count_synth = yes
	bst_count_steel_mill = yes
	bst_count_steel_mill_hydro = yes
	bst_count_aluminium_mill = yes
	bst_count_aluminium_mill_hydro = yes
}

bst_count_civf = {
	every_country = {
		set_variable = { This.bst_country_total_civf = 0 }
	}
	every_state = {
		OWNER = { add_to_variable = { This.bst_country_total_civf = PREV.industrial_complex_level } }
	}
}
