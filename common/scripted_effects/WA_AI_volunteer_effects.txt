### manages volunteers

WA_AI_update_volunteer_targets = {

	### volunteers

	if = {
		limit = {
			NOT = { has_war = yes num_divisions < 30 }

			OR = {
				AND = { has_government = neutrality NOT = { threat < 0.4 } }
				AND = { has_government = fascism }
				AND = { has_government = communism }
				AND = { has_government = democratic NOT = { threat < 0.5 } }
			}

			any_other_country = { has_war = yes }
		}

		every_country = { limit = { WA_AI_send_volunteers = yes }

			add_to_temp_array = { array = updated_volunteer_targets value = THIS.id }
		}

		# clear invalid targets
		for_each_scope_loop = { array = WA_AI_volunteer_targets

			if = { limit = { NOT = { is_in_array = { array = updated_volunteer_targets value = THIS.id } } }

				add_to_array = { array = ROOT.WA_AI_remove_volunteer value = THIS.id }
			}
		}
		for_each_scope_loop = { array = WA_AI_remove_volunteer

			WA_AI_clear_volunteer_target = yes
		}
		clear_array = WA_AI_remove_volunteer

		# set volunteer targets
		for_each_scope_loop = { array = updated_volunteer_targets

			if = { limit = { NOT = { is_in_array = { array = ROOT.WA_AI_volunteer_targets value = THIS.id } } }

				ROOT = { WA_AI_set_volunteer_target = yes }
			}
		}
	}

	### expeditionaries

	else_if = {
		limit = {
			has_war = yes
			is_major = no
			OR = { is_in_faction = yes is_subject = yes }
		}

		WA_AI_clear_volunteer_targets = yes

		every_country = { limit = { WA_AI_send_expeditionary = yes }

			add_to_temp_array = { array = updated_expeditionary_targets value = THIS.id }
		}

		# clear invalid targets
		for_each_scope_loop = { array = WA_AI_expeditionary_targets

			if = { limit = { NOT = { is_in_array = { array = updated_expeditionary_targets value = THIS.id } } }

				add_to_array = { array = ROOT.WA_AI_remove_expeditionary value = THIS.id }
			}
		}
		for_each_scope_loop = { array = WA_AI_remove_expeditionary

			WA_AI_clear_expeditionary_target = yes
		}
		clear_array = WA_AI_remove_expeditionary

		# set expeditionary targets
		for_each_scope_loop = { array = updated_expeditionary_targets

			if = { limit = { NOT = { is_in_array = { array = ROOT.WA_AI_expeditionary_targets value = THIS.id } } }

				ROOT = { WA_AI_set_expeditionary_target = yes }
			}
		}
	}

	else = {
		WA_AI_clear_volunteer_targets = yes
		WA_AI_clear_expeditionary_targets = yes
	}
}

###

WA_AI_set_volunteer_target = { # THIS = SENDER PREV = TARGET

	PREV = { set_country_flag = WA_AI_volunteers_from_@PREV }
	add_to_array = { array = WA_AI_volunteer_targets value = PREV.id }
	if = { limit = { ROOT = { has_country_flag = WA_AI_logging } } log = "[GetYear] [GetMonth] | AI | [This.GetName] | VOLUNTEERS: [Prev.GetName]" }
}

WA_AI_clear_volunteer_target = { # THIS = TARGET PREV = SENDER

	clr_country_flag = WA_AI_volunteers_from_@PREV

	# compare tags because the id may not match after civil wars
	for_each_loop = { array = PREV.WA_AI_volunteer_targets

		var:v = {
			if = { limit = { tag = PREV }

				remove_from_array = { array = PREV.PREV.WA_AI_volunteer_targets index = i }
				set_temp_variable = { break = 1 }
			}
		}
	}
	set_temp_variable = { break = 0 }

	if = { limit = { PREV = { has_country_flag = WA_AI_logging } } log = "[GetYear] [GetMonth] | AI | [Prev.GetName] | CANCEL VOLUNTEERS: [This.GetName]" }
}

WA_AI_clear_volunteer_targets = { # THIS = SENDER

	if = { limit = { check_variable = { WA_AI_volunteer_targets^num > 0 } }

		every_country = { limit = { has_country_flag = WA_AI_volunteers_from_@PREV }

			WA_AI_clear_volunteer_target = yes
		}

		clear_array = WA_AI_volunteer_targets
	}
}
