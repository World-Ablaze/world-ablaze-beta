					army_infantry_2

					base = 0
					modifier = {
						add = 100
						date > 1939.6.1
					}
				}
			}
		}
		AST_john_northcott = {
			add_skill_level = 2
			add_attack = 2
			add_defense = 2
			add_planning = 1
			add_logistics = 1
		}
		add_advisor_role = {
			character = "AST_john_northcott"
			advisor = {
				slot = high_command
				idea_token = AST_john_northcott
				ledger = army

				ai_will_do = {

					}
