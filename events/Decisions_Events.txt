	title = generic_purchase_ships_events.01.t
	desc = generic_purchase_ships_events.01.desc
	is_triggered_only = yes
	picture = GFX_report_event_merchant_ship_01
	# Yes
	option = {
		name = generic_purchase_ships_events.01.a
		ai_chance = {
			base = 0
			modifier = {
				add = 50
				is_in_faction_with = FROM
			}
			modifier = {
				add = 50
				has_same_ideology = yes
			}
			modifier = {
				add = 10
				has_opinion = {
					target = FROM
					value > 10
				}
			}
			modifier = {
				add = 20
				has_opinion = {
					target = FROM
					value > 40
				}
			}
			modifier = {
				add = 30
				has_opinion = {
					target = FROM
					value > 80
				}
			}
			modifier = {
				factor = 0
				OR = {
					has_war_with = FROM
					has_civil_war = yes
					has_capitulated = yes
					has_opinion = {
						target = FROM
						value < 0
					}
				}
			}
		}
		add_offsite_building = { type = industrial_complex  level = 1 }
		FROM = {
			country_event = {
				id = generic_purchase_ships_events.02
				days = 1
			}
		}
	}
	# No
	option = {
		name = generic_purchase_ships_events.01.b
		ai_chance = {
			base = 20
		}
		FROM = {
			country_event = {
				id = generic_purchase_ships_events.03
				days = 1
			}
		}
	}
}
