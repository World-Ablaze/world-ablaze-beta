	id = jap_armor.942
	hidden = yes

	mean_time_to_happen = { days = 1 }

	fire_only_once = yes

	trigger = {
		tag = JAP
		is_ai = yes
		threat > 0.1
		NOT = { has_war_with = USA }
	}

	immediate = {
		USA = {
			add_opinion_modifier = {
				target = JAP
				modifier = improved_relations_ai
			}
		}
	}

	option = {}

}
