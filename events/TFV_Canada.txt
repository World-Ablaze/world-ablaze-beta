# Canadian national focus.
country_event = {
	id = tfv_canada.30
	title = tfv_canada.30.t	# Mexican civil war starts.
	desc = tfv_canada.30.d	#
	picture = GFX_report_event_mex_civil_war

	fire_only_once = yes

	trigger = {
		original_tag = MEX
		has_defensive_war = no
		NOT = { has_government = fascism }
		has_idea = synarchist_support_surge
		fascism > 0.3
	}

	mean_time_to_happen = {
		months = 4
		modifier = {
			factor = 0.5
			fascism > 0.5
		}
	}

	option = { # Support Nationalists
		name = tfv_canada.30.a
		trigger = { is_ai = no }
		hidden_effect = {
			start_civil_war = {
				ruling_party = fascism
				ideology = neutrality
				size = 0.5
			}
			random_other_country = {
				limit = {
					original_tag = MEX
					has_government = neutrality
				}
				set_equipment_fraction = 0.3
				set_stability = 0.9
				set_war_support = 0.8
				add_popularity = {
					ideology = democratic
					popularity = 0.35
				}
			}
			news_event = { hours = 6 id = news.250 }

			set_equipment_fraction = 0.3
			set_stability = 0.9
			set_war_support = 0.8

			add_popularity = {
				ideology = fascism
				popularity = 1
			}

			remove_ideas = synarchist_support_surge
		}
		custom_effect_tooltip = mex_nationalists_chosen
	}
	option = { # Support Republicans
		name = tfv_canada.30.b
		hidden_effect = {
			start_civil_war = {
				ideology = fascism
				size = 0.5
			}
			random_other_country = {
				limit = {
					original_tag = MEX
					has_government = fascism
				}
				set_equipment_fraction = 0.3
				set_stability = 0.9
				set_war_support = 0.8
			}
			news_event = { hours = 6 id = news.250 }

			set_politics = {
				ruling_party = neutrality
				elections_allowed = no
			}
			add_popularity = {
				ideology = neutrality
				popularity = 0.5
			}
			add_popularity = {
				ideology = fascism
				popularity = -1
			}
			set_equipment_fraction = 0.3
			set_stability = 0.9
			set_war_support = 0.8

			remove_ideas = synarchist_support_surge
		}
		custom_effect_tooltip = mex_republicans_chosen
	}

	option = {
		name = tfv_canada.30.c # Peacefully let the fascists form the government

		trigger = {
			fascism > 0.5
		}

		ai_chance = {
			base = 5
		}

		set_politics = {
			ruling_party = fascism
			elections_allowed = no
		}
		remove_ideas = synarchist_support_surge
		add_ideas = shattered_government
	}
}
