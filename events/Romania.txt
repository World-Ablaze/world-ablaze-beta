# Soviet Union demands Bessarabia
country_event = {
	id = romania.1
	title = romania.1.t
	desc = romania.1.d
	picture = GFX_report_event_generic_read_write

	is_triggered_only = yes


	option = { # Accept demands
		
		name = romania.1.a
		ai_chance = {
			factor = 90
			modifier = { # Don't accept if in a faction that can protect them
				factor = 0
				OR = {
					any_other_country = {
						is_major = yes
						NOT = { tag = SOV }
						OR = {
							is_in_faction_with = ROM
							has_guaranteed = ROM
						}
					}
					has_idea = anti_soviet_pact
					has_government = fascism
				}
			}
		}
		SOV = {
			country_event = { days = 1 id = soviet.4 }
		}
		if = {
			limit = {
				has_government = neutrality
			}
			add_popularity = { ideology = neutrality popularity = -0.30 }
		}
		if = {
			limit = {
				has_government = democratic
			}
			add_popularity = { ideology = democratic popularity = -0.30 }
		}
		if = {
			limit = {
				has_government = communism
			}
			add_popularity = { ideology = communism popularity = -0.30 }
		}
		if = {
			limit = {
				has_government = fascism
			}
			add_popularity = { ideology = fascism popularity = -0.30 }
			add_stability = -0.8
			add_war_support = -0.8 #subject to change
		}
		set_country_flag = bessarabia_ceded
	}
	option = { # Refuse
		name = romania.1.b
		ai_chance = {
			base = 0
			modifier = {
				add = 10
				NOT = {
					any_other_country = {
						is_major = yes
						NOT = { tag = SOV }
						OR = {
							is_in_faction_with = ROM
							has_guaranteed = ROM
						}
					}
					has_idea = anti_soviet_pact
					has_government = fascism
				}
			}
		}
		SOV = {
			country_event = { days = 1 id = soviet.5 }
			add_opinion_modifier = {
				target = ROM
				modifier = refused_bessarabia
			}
			effect_tooltip = {
				create_wargoal = {
					type = take_state
					target = ROM
					generator = { 80 78 766 }
				}
			}
		}
		if = {
			limit = {
				ROM = {
					is_in_faction_with = GER
				}
			}
			ROM = {
				add_opinion_modifier = {
					target = GER
					modifier = refused_support
				}
				activate_mission = activate_decision
			}
		}
		if = {
			limit = {
				ROM = {
					NOT = {
						has_government = fascism
					}
				}	
			}
			GER = {
				remove_from_faction = ROM
				add_opinion_modifier = {
					target = ROM
					modifier = endangered_oilfields
				}
				create_wargoal = {
					type = annex_everything
					target = ROM
				}
				set_country_flag = romania_endangered_oilfields
			}
		}
	}
}

#Soviet response to Romanian surrender
country_event = {
	id = romania.8
	title = romania.8.t
	desc = romania.8.d
	picture = GFX_report_event_romanian_soldiers

	is_triggered_only = yes

	#Accept surrender
	option = {
		name = romania.8.a
		army_experience = 25
		ai_chance = {
			base = 9
			modifier = {
				factor = 0
				has_completed_focus = SOV_a_global_class_struggle
			}
		}
		white_peace = ROM
		transfer_state = 80
		transfer_state = 78
		transfer_state = 766
		ROM = {
			add_ai_strategy = {
				type = antagonize
				id = "SOV"
				value = 200
			}
		}
		if = {
			limit = {
				GER = {
					is_subject = no
					NOT = { is_in_faction_with = SOV }
				}
			}
			add_ai_strategy = {
				type = alliance
				id = "GER"
				value = 2000
			}
			add_ai_strategy = {
				type = befriend
				id = "GER"
				value = 200
			}
			GER = {
				add_ai_strategy = {
					type = alliance
					id = "ROM"
					value = 2000
				}
			}
		}
	}

	#Take it all
	option = {
		name = finland.8.b
		ai_chance = {
			base = 1
			modifier = {
				factor = 0
				has_completed_focus = SOV_great_purge
			}
		}
		ROM = { country_event = { id = romania.9 } }
	}
}
