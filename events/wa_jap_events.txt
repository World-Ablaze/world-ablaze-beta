country_event = {
	id = jap_armor.421
	hidden = yes

	mean_time_to_happen = { days = 1 }

	fire_only_once = yes

	trigger = {
		tag = RNC
		NOT = { has_war_with = CHI }
		JAP = { has_completed_focus = JAP_the_china_project }
		CHI = { is_subject_of = JAP }
	}

	immediate = {
		if = {
			limit = {
				country_exists = CHI
				CHI = { is_subject_of = JAP }
			}
			RNC = { annex_country = { target = CHI } }
		}
		if = {
			limit = {
				JAP = { is_ai = yes }
			}
			every_state = {
				limit = {
					OR = {
						is_fully_controlled_by = JAP
					}
					is_core_of = CHI
					NOT = { state = 745 }
					NOT = { state = 524 }
				}
				RNC = { transfer_state = PREV }
			}
		}
	}

	option = {}

}
